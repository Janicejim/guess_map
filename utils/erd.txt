
# Visualize on https://quick-erd.surge.sh
users
----
id pk
name varchar(64) null
email varchar(100)not null
password varchar(200) not null
profile_image varchar(255) null
description text null
role enum ('admin','user')
created_at timestamp
updated_at timestamp



game
----
id pk
user_id integer fk >- users.id
media varchar(200)
target_location point 
hints_1 text
hints_2 text
answer_name varchar(255)
answer_address varchar(255)
answer_description text
status enum('active','inactive','completed')
created_at timestamp
updated_at timestamp


like_dislike
----
id pk
game_id integer fk >- game.id
user_id integer fk >- users.id
type enum ('like','dislike')
created_at timestamp
updated_at timestamp


game_history
----
id pk
game_id integer fk >- game.id
user_id integer fk >- users.id
attempts integer
is_win boolean
created_at timestamp
updated_at timestamp

store_record
----
id pk
game_id integer fk >- game.id
user_id integer fk >- users.id
amount_change integer
created_at timestamp
updated_at timestamp


score_description
----
id pk
description text
score_change integer
image varchar(255)
created_at timestamp
updated_at timestamp


score_record
----
id pk
user_id integer fk >- users.id
score_change integer
score_description_id integer fk >- score_description.id
created_at timestamp
updated_at timestamp

award
----
id pk
name varchar(255)
image varchar (255)
score integer
quantity integer
status enum('active','inactive')
created_at timestamp
updated_at timestamp


user_award
----
id pk
award_id integer fk >- award.id
user_id integer fk >- users.id
score integer
created_at timestamp
updated_at timestamp


collection
----
id pk
user_id integer fk >- users.id
game_id integer fk >- game.id
status enum('active','inactive')
created_at timestamp
updated_at timestamp

check_in
----
id pk
user_id integer fk >- users.id
game_id integer fk >- game.id
status enum('active','inactive')
image varchar(255)
message varchar(255)
created_at timestamp
updated_at timestamp
